    ```swift

    import AppKit
    import SwiftUI

    struct AVPlayerViewRepresentable: NSViewRepresentable {
        let url: URL
        func makeNSView(context: Context) -> AVPlayerView {
            let view = AVPlayerView()
            view.player = AVPlayer(url: url)
            view.controlsStyle = .default
            view.player?.play()
            return view
        }
        func updateNSView(_ nsView: AVPlayerView, context: Context) {}
    ```

    Update ContentView.swift to navigate to ChunkReviewView when the folder scan returns a .success result.

    Current ContentView just mounts FolderPickerView with its own Store. The navigation pattern:
    - Show FolderPickerView initially
    - When scanResult is .success(chunks), show ChunkReviewView

    Use a root AppFeature TCA reducer OR a simple @State navigation approach. Given the two-screen flow is simple and Phase 4 adds a third screen, use a root TCA AppFeature that holds both child reducers via Scope/ifLet. However, for velocity, a simpler approach is acceptable: hold the navigation state in ContentView using @State var scannedChunks: [ScannedChunk]? and pass the store result up via a callback/delegate action.

    Simplest correct approach — update ContentView to hold an AppFeature TCA root reducer:

    ```swift
    // AppFeature.swift (new file)
    @Reducer
    struct AppFeature {
        @ObservableState
        struct State: Equatable {
            var folderPicker = FolderPickerFeature.State()
            var chunkReview: ChunkReviewFeature.State? = nil
        }
        enum Action {
            case folderPicker(FolderPickerFeature.Action)
            case chunkReview(ChunkReviewFeature.Action)
        }
        var body: some ReducerOf<Self> {
            Scope(state: \.folderPicker, action: \.folderPicker) { FolderPickerFeature() }
            Reduce { state, action in
                switch action {
                case let .folderPicker(.scanCompleted(.success(chunks))):
                    state.chunkReview = ChunkReviewFeature.State(chunks: chunks)
                    return .none
                case .chunkReview:
                    return .none
                case .folderPicker:
                    return .none
                }
            }
            .ifLet(\.chunkReview, action: \.chunkReview) { ChunkReviewFeature() }
        }
    }
    ```

    ContentView.swift:
    ```swift
    struct ContentView: View {
        let store: StoreOf<AppFeature>

        var body: some View {
            if store.chunkReview != nil {
                ChunkReviewView(store: store.scope(state: \.chunkReview!, action: \.chunkReview))
            } else {
                FolderPickerView(store: store.scope(state: \.folderPicker, action: \.folderPicker))
            }
        }
    }
    ```

    Update the @main App entry point (GoProStitcherApp.swift) to initialize Store(initialState: AppFeature.State()) { AppFeature() } and pass it to ContentView.

    Create GoProStitcher/AppFeature.swift as a new file. Add to project.yml sources (or rely on glob).

    IMPORTANT: Check GoProStitcherApp.swift first to see how the window/scene is set up before editing ContentView. The store must be created at the App level and passed down, not created inside ContentView (avoids recreation on rerender).
