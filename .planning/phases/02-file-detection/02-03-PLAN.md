    **FolderPickerFeature.swift:**

    Import ComposableArchitecture and GoProStitcherKit.

    ```swift
    @Reducer
    public struct FolderPickerFeature {
        @ObservableState
        public struct State: Equatable {
            var scanResult: FolderScanResult? = nil    // nil = no folder selected yet
            var isLoading: Bool = false
        }

        public enum Action {
            case selectFolderButtonTapped
            case folderSelected(URL)
            case scanCompleted(FolderScanResult)
        }

        public var body: some Reducer<State, Action> {
            Reduce { state, action in
                switch action {
                case .selectFolderButtonTapped:
                    state.isLoading = true
                    return .run { send in
                        // Present NSOpenPanel on the main thread
                        let url = await MainActor.run {
                            let panel = NSOpenPanel()
                            panel.canChooseDirectories = true
                            panel.canChooseFiles = false
                            panel.allowsMultipleSelection = false
                            panel.prompt = "Select GoPro Folder"
                            guard panel.runModal() == .OK else { return nil as URL? }
                            return panel.url
                        }
                        if let url {
                            await send(.folderSelected(url))
                        } else {
                            // User cancelled — reset loading
                            await send(.scanCompleted(.empty))  // treat cancel as no selection; handled below
                        }
                    }

                case .folderSelected(let url):
                    return .run { send in
                        let result = FolderScanner.scan(url: url)
                        await send(.scanCompleted(result))
                    }

                case .scanCompleted(let result):
                    state.isLoading = false
                    state.scanResult = result
                    return .none
                }
            }
        }
    }
    ```

    Note: If user cancels the panel, set scanResult back to nil (not .empty). Adjust the cancel path:
    ```swift
    // On cancel: don't overwrite previous result — send a dedicated cancel action or use nil
    ```
    Simplest approach: add `case userCancelledPicker` action that just sets `state.isLoading = false` without changing `scanResult`. Replace the `.empty` fallback above with `.userCancelledPicker`.
