**1b. Create HomeFeature.swift**

Create `GoProStitcher/Features/Home/HomeFeature.swift`:

```swift
import ComposableArchitecture

@Reducer
struct HomeFeature {
    @ObservableState
    struct State: Equatable {}

    enum Action {
        case stitchVideoTapped
        case extractAudioTapped
    }

    var body: some ReducerOf<Self> {
        Reduce { state, action in
            .none  // Navigation handled by parent AppFeature
        }
    }
}
```

**1c. Create HomeView.swift**

Create `GoProStitcher/Features/Home/HomeView.swift` with an extensible `ToolDescriptor` approach so adding a third tool requires only one new entry in the array:

```swift
import ComposableArchitecture
import SwiftUI

struct ToolDescriptor {
    let title: String
    let subtitle: String
    let systemImage: String
    let action: AppFeature.Action
}

struct HomeView: View {
    let store: StoreOf<AppFeature>

    private let tools: [ToolDescriptor] = [
        ToolDescriptor(
            title: "Stitch Video",
            subtitle: "Combine GoPro chapter files into one",
            systemImage: "film.stack",
            action: .home(.stitchVideoTapped)
        ),
        ToolDescriptor(
            title: "Extract Audio",
            subtitle: "Pull audio track from any MP4",
            systemImage: "waveform",
            action: .home(.extractAudioTapped)
        ),
    ]

    var body: some View {
        VStack(spacing: 24) {
            Text("GoPro Toolkit")
                .font(.title)
                .fontWeight(.bold)
                .padding(.top, 32)

            ForEach(tools, id: \.title) { tool in
                Button {
                    store.send(tool.action)
                } label: {
                    HStack(spacing: 16) {
                        Image(systemName: tool.systemImage)
                            .font(.system(size: 28))
                            .frame(width: 44)
                        VStack(alignment: .leading, spacing: 4) {
                            Text(tool.title)
                                .font(.headline)
                            Text(tool.subtitle)
                                .font(.caption)
                                .foregroundStyle(.secondary)
                        }
                        Spacer()
                        Image(systemName: "chevron.right")
                            .foregroundStyle(.secondary)
                    }
                    .padding()
                    .background(.regularMaterial, in: RoundedRectangle(cornerRadius: 12))
                }
                .buttonStyle(.plain)
            }

            Spacer()
        }
        .padding(.horizontal, 24)
        .frame(minWidth: 400, minHeight: 300)
    }
}
```

Note: `HomeView` takes a `StoreOf<AppFeature>` (not `StoreOf<HomeFeature>`) so it can dispatch top-level navigation actions directly without requiring a separate store scope.

**2a. Refactor AppFeature.swift**

Replace the `showAudioPicker: Bool` flag with an `enum ActiveTool` and add a `home` child reducer scope. The complete rewrite:

- Add inside `AppFeature`:
  ```swift
  enum ActiveTool: Equatable { case stitch, audio }
  ```

- In `State`, replace `var showAudioPicker: Bool = false` with `var activeTool: ActiveTool? = nil`. Keep all other state fields.

- In `Action`, replace `case showAudioPickerTapped` with:
  ```swift
  case home(HomeFeature.Action)
  case backToHome
  ```

- Add a `Scope` for home at the top of `body`:
  ```swift
  Scope(state: \.home, action: \.home) { HomeFeature() }
  ```
  (Add `var home = HomeFeature.State()` to `State`.)

- In `Reduce`, replace the `showAudioPickerTapped` case with:
  ```swift
  case .home(.stitchVideoTapped):
      state.activeTool = .stitch
      return .none

  case .home(.extractAudioTapped):
      state.activeTool = .audio
      return .none

  case .home:
      return .none

  case .backToHome:
      state.activeTool = nil
      state.chunkReview = nil
      state.stitchProgress = nil
      state.audioPicker = AudioFilePickerFeature.State()
      state.audioExtraction = nil
      state.folderPicker = FolderPickerFeature.State()
      return .none
  ```

- Update the `audioPicker(.fileSelected)` case: remove `state.showAudioPicker = false` (it no longer exists).
- Update the `audioPicker(.userCancelledPicker)` case: instead of `state.showAudioPicker = false`, send `.backToHome` or set `state.activeTool = nil`.

**2b. Update ContentView.swift**

Replace the entire body with a switch on `store.activeTool`:

```swift
var body: some View {
    switch store.activeTool {
    case .none:
        HomeView(store: store)

    case .stitch:
        if store.stitchProgress != nil {
            StitchProgressView(
                store: store.scope(state: \.stitchProgress!, action: \.stitchProgress)
            )
        } else if store.chunkReview != nil {
            ChunkReviewView(
                store: store.scope(state: \.chunkReview!, action: \.chunkReview)
            )
        } else {
            VStack {
                Button("< Back") { store.send(.backToHome) }
                    .padding(.top, 8)
                    .frame(maxWidth: .infinity, alignment: .leading)
                FolderPickerView(
                    store: store.scope(state: \.folderPicker, action: \.folderPicker)
                )
            }
            .padding(.horizontal, 16)
        }

    case .audio:
        if store.audioExtraction != nil {
            AudioExtractionView(
                store: store.scope(state: \.audioExtraction!, action: \.audioExtraction)
            )
        } else {
            // Show file picker immediately; cancelling returns to home
            AudioFilePickerView(
                store: store.scope(state: \.audioPicker, action: \.audioPicker)
            )
        }
    }
}
```

The stitch sub-tree also needs back-to-home reachability when the user is at the FolderPicker screen (shown above). The ChunkReview and StitchProgress screens do not need a visible back button for this phase — their existing dismiss flows remain unchanged.
