    3. Add DesignTokenComplianceTests class to GoProStitcherIntegrationTests/DesignSystemTests.swift.
    Append to the existing file:

    ```swift
    // MARK: - Design Token Compliance Tests
    // These tests read source files at runtime to verify no system styling tokens remain.
    // They fail fast if a view is accidentally restyled with system fonts/colors.

    final class DesignTokenComplianceTests: XCTestCase {

        private let bannedPatterns: [String] = [
            "ProgressView(",
            "RoundedRectangle(cornerRadius",
            ".shadow(",
            ".blur(",
            "regularMaterial",
            "windowBackgroundColor",
            "font(.title",
            "font(.headline",
            "font(.subheadline",
            "font(.body",
            "font(.caption",
            "font(.largeTitle",
            "fontWeight(.bold",
            "fontWeight(.semibold",
            ".foregroundStyle(.secondary",
            ".foregroundColor(.green",
            ".foregroundColor(.red",
            ".easeInOut",
        ]

        private let viewFiles: [String] = [
            "Features/Home/HomeView.swift",
            "Features/FolderPicker/FolderPickerView.swift",
            "Features/ChunkReview/ChunkReviewView.swift",
            "Features/ChunkReview/ChunkPreviewModal.swift",
            "Features/StitchProgress/StitchProgressView.swift",
            "Features/AudioExtraction/AudioExtractionView.swift",
            "ContentView.swift",
            "GoProStitcherApp.swift",
        ]

        func test_noSystemStylingInViewFiles() throws {
            // Locate the GoProStitcher source directory relative to the test bundle
            let bundle = Bundle(for: DesignTokenComplianceTests.self)
            guard let bundlePath = bundle.bundleURL
                    .deletingLastPathComponent()  // Contents/MacOS
                    .deletingLastPathComponent()  // Contents
                    .deletingLastPathComponent()  // GoProStitcher.app
                    .deletingLastPathComponent()  // Debug
                    .deletingLastPathComponent()  // build/
                    .appendingPathComponent("GoProStitcher")
                    .path
                    .nonEmpty else {
                // If path resolution fails in CI, skip gracefully
                throw XCTSkip("Cannot resolve source path from test bundle — skipping compliance check")
            }

            var violations: [String] = []
            for relPath in viewFiles {
                let fullPath = (bundlePath as NSString).appendingPathComponent(relPath)
                guard let source = try? String(contentsOfFile: fullPath, encoding: .utf8) else {
                    violations.append("MISSING: \(relPath)")
                    continue
                }
                for pattern in bannedPatterns {
                    if source.contains(pattern) {
                        violations.append("\(relPath): found banned token '\(pattern)'")
                    }
                }
            }

            XCTAssertTrue(violations.isEmpty,
                "Design token compliance violations found:\n" + violations.joined(separator: "\n"))
        }
    }

    private extension String {
        var nonEmpty: String? { isEmpty ? nil : self }
    }
    ```
