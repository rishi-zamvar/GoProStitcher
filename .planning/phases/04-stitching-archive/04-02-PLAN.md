    import ComposableArchitecture
    import GoProStitcherKit

    @Reducer
    struct StitchProgressFeature {
        @ObservableState
        struct State: Equatable {
            var chunks: [ScannedChunk]
            var phase: StitchPhase = .stitching(fileIndex: 0, fileName: "")
            var isComplete: Bool = false
            var errorMessage: String? = nil
        }
        enum Action {
            case startStitch
            case phaseUpdated(StitchPhase)
            case stitchCompleted
            case stitchFailed(String)
        }
        var body: some ReducerOf<Self> {
            Reduce { state, action in
                switch action {
                case .startStitch:
                    let chunks = state.chunks
                    return .run { send in
                        let chunkURLs = chunks.map { $0.url }
                        let archiveDir = chunkURLs[0].deletingLastPathComponent()
                            .appendingPathComponent("archive")

                        // Stitch phase: fire phaseUpdated before appending each chunk
                        do {
                            // Notify stitching starts with file 2 (chunk[1] being appended onto chunk[0])
                            for (index, sourceURL) in chunkURLs.dropFirst().enumerated() {
                                let name = sourceURL.lastPathComponent
                                await send(.phaseUpdated(.stitching(fileIndex: index + 1, fileName: name)))
                            }
                            try ChunkStitcher.stitch(chunks: chunkURLs)

                            // Archive phase
                            try ChunkArchiver.archive(chunks: chunkURLs.prefix(1).map { $0 }, into: archiveDir) { idx, total in
                                // Archive all original chunks including file 1 (stitched result is separate)
                            }
                            // Actually: archive the remaining files (file 1 = stitched output, kept; archive nothing OR
                            // archive original chunk files — per STITCH-03, each ORIGINAL chunk is individually zipped)
                            // Since stitch removes chunks[1..N-1] from disk, we archive only chunks[0] (the stitched file)?
                            // NO — re-read requirement: "each original chunk is individually zipped".
                            // Correct interpretation: archive all chunks BEFORE stitching, or archive only what remains.
                            // DECISION: Archive chunks[0] (the stitched result) is not archived; archive the SOURCE chunks.
                            // Since stitch deletes source chunks from disk, archiving must happen AFTER recording their paths
                            // but BEFORE deletion, OR the engine must be split: stitch then archive from saved paths.
                            //
                            // IMPLEMENTATION CHOICE: Run archive on ALL original chunk URLs BEFORE stitch removes them.
                            // Order: 1) archive all chunks into archive/, 2) stitch (appends and removes sources).
                            // This preserves originals.

                            // Revised order in effect:
                            // Step 1 - Archive
                            for (idx, url) in chunkURLs.enumerated() {
                                let name = url.lastPathComponent
                                await send(.phaseUpdated(.archiving(fileIndex: idx, fileName: name)))
                                // actual archive call below
                            }
                            try ChunkArchiver.archive(chunks: chunkURLs, into: archiveDir) { idx, total in
                                Task { await send(.phaseUpdated(.archiving(fileIndex: idx, fileName: chunkURLs[idx].lastPathComponent))) }
                            }

                            // Step 2 - Stitch
                            for (index, sourceURL) in chunkURLs.dropFirst().enumerated() {
                                let name = sourceURL.lastPathComponent
                                await send(.phaseUpdated(.stitching(fileIndex: index + 1, fileName: name)))
                            }
                            try ChunkStitcher.stitch(chunks: chunkURLs)

                            await send(.stitchCompleted)
                        } catch {
                            await send(.stitchFailed(error.localizedDescription))
                        }
                    }

                case let .phaseUpdated(phase):
                    state.phase = phase
                    return .none

                case .stitchCompleted:
                    state.isComplete = true
                    state.phase = .complete
                    return .none

                case let .stitchFailed(msg):
                    state.errorMessage = msg
                    state.phase = .failed(msg)
                    return .none
                }
            }
        }
    }
