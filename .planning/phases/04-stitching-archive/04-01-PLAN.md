<feature>
  <name>ChunkStitcher — sequential binary append</name>
  <files>
    GoProStitcherKit/Sources/GoProStitcherKit/ChunkStitcher.swift
    GoProStitcherKit/Tests/GoProStitcherKitTests/ChunkStitcherTests.swift
  </files>
  <behavior>
    ChunkStitcher.stitch(chunks: [URL]) appends each chunk (index 1..N-1) onto chunk 0 in-place using FileHandle.
    - chunks[0] is the destination: opened for writing, seeked to end before each append
    - chunks[1..N-1] are read in 1 MB read-buffers (never load whole file into memory)
    - After append, the source file (chunks[1..N-1]) is removed from disk
    - No intermediate copy files are created

    Error cases:
    - ChunkStitcherError.destinationNotFound if chunks[0] does not exist
    - ChunkStitcherError.sourceNotFound(url) if any chunk[1..N-1] does not exist
    - ChunkStitcherError.empty if chunks array is empty or has only 1 element (no-op or error — stitch(chunks:) with count < 2 should throw .empty)

    Test cases (input -> expected output):
    - stitch([a(10B), b(20B), c(30B)]) -> file a is 60 bytes, bytes equal a+b+c concatenated; b and c removed
    - stitch([a(0B)]) -> throws .empty
    - stitch([]) -> throws .empty
    - stitch([nonexistent, b]) -> throws .destinationNotFound
    - stitch([a, nonexistent]) -> throws .sourceNotFound(url)
  </behavior>
  <implementation>
    public enum ChunkStitcher {
        public static func stitch(chunks: [URL]) throws { ... }
    }
    public enum ChunkStitcherError: Error { case empty, destinationNotFound, sourceNotFound(URL) }

    Implementation: FileHandle(forWritingAtPath:) for destination, seekToEndOfFile(). For each source chunk: FileHandle(forReadingAtPath:), read 1 MB at a time, write to destination handle, close source handle, FileManager.removeItem(at: sourceURL).

    Use 1 MB buffer constant: let bufferSize = 1_048_576
  </implementation>
</feature>

<feature>
  <name>ChunkArchiver — individual zip per chunk</name>
  <files>
    GoProStitcherKit/Sources/GoProStitcherKit/ChunkArchiver.swift
    GoProStitcherKit/Tests/GoProStitcherKitTests/ChunkArchiveTests.swift
  </files>
  <behavior>
    ChunkArchiver.archive(chunks: [URL], into archiveDir: URL) creates archiveDir if needed, then for each chunk URL creates a zip at archiveDir/<filename>.zip using the system /usr/bin/zip command via Process.

    Progress callback: archive(chunks:into:progress:) where progress: (Int, Int) -> Void receives (currentIndex, totalCount).

    Error cases:
    - ChunkArchiverError.sourceNotFound(url) if chunk does not exist
    - ChunkArchiverError.zipFailed(url, exitCode) if zip process exits non-zero

    Test cases:
    - archive([a, b, c], into: archiveDir) -> archiveDir/GH010001.zip, archiveDir/GH020001.zip, archiveDir/GH030001.zip all exist; each zip contains the original file
    - archive([nonexistent], into: dir) -> throws .sourceNotFound
    - progress callback called once per chunk with correct indices (0, 1, 2) and total (3)
  </behavior>
  <implementation>
    public enum ChunkArchiver {
        public static func archive(chunks: [URL], into archiveDir: URL, progress: ((Int, Int) -> Void)? = nil) throws { ... }
    }
    public enum ChunkArchiverError: Error { case sourceNotFound(URL), zipFailed(URL, Int) }

    Use Process to invoke /usr/bin/zip:
      let process = Process()
      process.executableURL = URL(fileURLWithPath: "/usr/bin/zip")
      process.arguments = ["-j", archiveDir.appendingPathComponent(chunkURL.lastPathComponent + ".zip").path, chunkURL.path]
      // -j: junk paths (store just filename, not full path)
      process.launch()
      process.waitUntilExit()
      if process.terminationStatus != 0 { throw .zipFailed(...) }
  </implementation>
</feature>
